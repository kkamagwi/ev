pip install virtualenv

pip install virtualenvwrapper-win


# Команда и ее Описание
* mkvirtualenv env-name	
    * Создаем новое окружение
* workon	
    * Смотрим список окружений
* workon env-name	
    * Меняем окружение
* deactivate	
    * Выходим из окружения
* rmvirtualenv env-name	
    * Удаляем окружение
